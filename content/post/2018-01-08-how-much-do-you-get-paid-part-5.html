Getting started

Load the data and the libraries as we’ve done before.

library(readr)
library(knitr)
library(kableExtra)
library(ggplot2)
library(dplyr)

# Read the data into R
dataSource <- "https://raw.githubusercontent.com/jacquietran/soccerSalariesDataPublic/master/soccerSalariesDataPublic.csv"
soccerSalariesData <- read_delim(dataSource, delim = ",", col_names = TRUE, col_types = NULL)

Salaries by highest qualification

# Set factor levels
soccerSalariesData$highestQualCategory <- factor(
    soccerSalariesData$highestQualCategory,
    levels = c("PhD", "Postgraduate", "Undergraduate", "Unclear"))

# Create a list to store common theme options for use across plots
plotFeatures <- theme(
    plot.title = element_text(size = 16),
    plot.subtitle = element_text(size = 14),
    axis.title.x = element_blank(),
  axis.text.x = element_text(size = 11),
    axis.title.y = element_text(size = 16),
    axis.text.y = element_text(size = 11),
    legend.position = "none")

p <- ggplot(soccerSalariesData, aes(x = highestQualCategory, y = salaryGBP))
p <- p + geom_boxplot(outlier.shape = NA, coef = 0)
p <- p + geom_point(alpha = 1/4, size = 4, aes(colour = highestQualCategory))
p <- p + scale_y_continuous(
    limits = c(0, 50000),
    breaks = seq(0, 50000, by = 10000))
p <- p + labs(
    title = "UK soccer salaries of support staff",
    subtitle = "By highest qualification",
    y = "Salary (GBP)")
p <- p + plotFeatures
salaryByHighestQual <- p

# Display the plot
salaryByHighestQual
